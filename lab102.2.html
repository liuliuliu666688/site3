onload = sortByFirstName; let sortByFirstName = function() { const tbl = document.getElementById("nameTable"); const rows = Array.from(tbl.rows).slice(1); // Skip header row // Sort rows by first name (assuming first column is first name) rows.sort((a, b) => { const nameA = a.cells[0].innerHTML.toLowerCase(); const nameB = b.cells[0].innerHTML.toLowerCase(); return nameA.localeCompare(nameB); }); // Remove existing rows (except header) while (tbl.rows.length > 1) { tbl.deleteRow(1); } // Append sorted rows rows.forEach(row => tbl.appendChild(row)); }
